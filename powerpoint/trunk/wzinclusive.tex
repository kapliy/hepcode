\slide{W,Z selections in muon channel}
{
Common selection:
\iteb
\item Single muon trigger: EF\_mu18\_MG or EF\_mu18\_MG\_medium
\item Primary vertex with $\ge$ 3 tracks
\item Reject events with $BadLooser$ jets or jets in $LarHole$ region
\item Combined STACO muons passing MCP quality cuts and $|z_0|~<~10~mm$
\item Muon kinematics: $p_T~>~20~\GeV$ and $|\eta|~<~2.4$
\item Tracking isolation: $\sum \pt(\Delta R~<~0.4) / \pt~<~0.1$
\itee
\Wmn~channel:
\iteb
\item Veto events with a 2nd isolated muon
\item $m_{T}^{W}~>~40~GeV$ and $E_{T}^{miss}~>~25~GeV$
\itee
\Zmm~channel:
\iteb
\item Veto events with a 3rd isolated muon
\item Opposite charges
\item $66~<~m_{\mu\mu}~<~116~GeV$
\itee

}
